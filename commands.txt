git clone - склонировал из github с HTTPS url 
git status - показывает изменненые данные что где на каком этапе, которые нуждаются в направлении 
git add - Добавляет/готовит файл для последующего коммита
git commit -m "описание изменения" - Сохраняте в истории версии
git branch -a - показывает все существующие ветви 
git switch "имя ветки" - переключается на ветку которую указали 
clear - чистит коммандную строку
ls -lah - показывает файлы
cd "указать путь" - переходить по папкам 
git log - показывает истории коммитов
git push - загрузка изменения на github
